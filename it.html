text = empty;
LONG = 1;
SHORT = 0;

function key(notes) {

    for each (note in notes) {
   
       if (note is long)
           text.append(LONG);
       else if (note is short)
           text.append(SHORT);
       else if (note is silent)
           text = text;
         
    }
	
	return text;
	
}
